🧑‍💻 6. MINIMAL PLAYABLE PROTOTYPE

Here’s your starting skeleton (Phaser):

import Phaser from "phaser";

class GameScene extends Phaser.Scene {
  constructor() {
    super("GameScene");
  }

  preload() {
    this.load.image("player", "assets/player.png");
    this.load.image("ladder", "assets/ladder.png");
  }

  create() {
    this.player = this.physics.add.sprite(200, 500, "player");
    this.player.setCollideWorldBounds(true);

    this.cursors = this.input.keyboard.createCursorKeys();

    this.ladders = this.physics.add.staticGroup();

    for (let i = 0; i < 20; i++) {
      this.ladders.create(
        Phaser.Math.Between(100, 300),
        600 - i * 100,
        "ladder"
      );
    }

    this.physics.add.collider(this.player, this.ladders);
  }

  update() {
    if (this.cursors.left.isDown) {
      this.player.setVelocityX(-160);
    } else if (this.cursors.right.isDown) {
      this.player.setVelocityX(160);
    } else {
      this.player.setVelocityX(0);
    }

    // Auto climb
    this.player.y -= 1;
  }
}

const config = {
  type: Phaser.AUTO,
  width: 400,
  height: 700,
  physics: { default: "arcade" },
  scene: GameScene
};

new Phaser.Game(config);

import Phaser from "phaser";

const FORM_TEXTURES = {
  intern: "player_intern",
  analyst: "player_analyst",
  manager: "player_manager",
  executive: "player_executive",
  entity: "player_entity"
};

export default class Player extends Phaser.Physics.Arcade.Sprite {
  constructor(scene, x, y) {
    super(scene, x, y, "player_intern");

    scene.add.existing(this);
    scene.physics.add.existing(this);

    this.setCollideWorldBounds(false);
    this.setGravityY(0);
    this.setDepth(10);

    this.form = "intern";
    this.demotionsRemaining = 3;
    this.score = 0;
    this.title = "Intern (Unpaid)";
  }

  moveLeft() {
    this.setVelocityX(-140);
  }

  moveRight() {
    this.setVelocityX(140);
  }

  stopX() {
    this.setVelocityX(0);
  }

  jump() {
    this.setVelocityY(-260);
  }

  climbTick(amount = 1.25) {
    this.y -= amount;
    this.score += amount;
  }

  setCareerState({ title, form }) {
    this.title = title;

    if (form !== this.form) {
      this.form = form;
      this.setTexture(FORM_TEXTURES[form]);
    }
  }

  demote() {
    this.demotionsRemaining -= 1;
  }
}

LadderManager.js
import Phaser from "phaser";
import { GAME_WIDTH, GAME_HEIGHT, LADDER_SPACING_Y } from "../constants.js";

export default class LadderManager {
  constructor(scene) {
    this.scene = scene;
    this.group = scene.physics.add.staticGroup();
    this.lastY = GAME_HEIGHT;
  }

  createInitial() {
    for (let i = 0; i < 12; i++) {
      this.spawnSegment(GAME_HEIGHT - i * LADDER_SPACING_Y);
    }
  }

  spawnSegment(y) {
    const x = Phaser.Math.Between(80, GAME_WIDTH - 80);
    const ladder = this.group.create(x, y, "ladder");
    ladder.refreshBody();
    this.lastY = y;
  }

  update(cameraY) {
    const topThreshold = cameraY - 200;

    this.group.children.iterate((child) => {
      if (child && child.y > cameraY + GAME_HEIGHT + 100) {
        child.destroy();
      }
    });

    while (this.lastY > topThreshold) {
      this.spawnSegment(this.lastY - LADDER_SPACING_Y);
    }
  }
}
GameScene.js
import Phaser from "phaser";
import Player from "../entities/Player.js";
import LadderManager from "../managers/LadderManager.js";
import CareerManager from "../managers/CareerManager.js";
import { GAME_WIDTH, GAME_HEIGHT, SCROLL_SPEED } from "../constants.js";

export default class GameScene extends Phaser.Scene {
  constructor() {
    super("GameScene");
  }

  create() {
    this.cursors = this.input.keyboard.createCursorKeys();

    this.player = new Player(this, GAME_WIDTH / 2, GAME_HEIGHT - 120);
    this.ladderManager = new LadderManager(this);
    this.careerManager = new CareerManager(this.player);

    this.ladderManager.createInitial();

    this.physics.add.collider(this.player, this.ladderManager.group);

    this.cameras.main.startFollow(this.player, false, 0.08, 0.08);

    this.events.emit("career-update", {
      title: this.player.title,
      demotionsRemaining: this.player.demotionsRemaining,
      score: this.player.score
    });
  }

  update() {
    if (this.cursors.left.isDown) {
      this.player.moveLeft();
    } else if (this.cursors.right.isDown) {
      this.player.moveRight();
    } else {
      this.player.stopX();
    }

    if (Phaser.Input.Keyboard.JustDown(this.cursors.space)) {
      this.player.jump();
    }

    this.player.climbTick(SCROLL_SPEED);
    this.careerManager.update();
    this.ladderManager.update(this.cameras.main.scrollY);

    if (this.player.y > this.cameras.main.scrollY + GAME_HEIGHT + 80) {
      this.handleDemotion();
    }

    this.events.emit("career-update", {
      title: this.player.title,
      demotionsRemaining: this.player.demotionsRemaining,
      score: Math.floor(this.player.score)
    });
  }

  handleDemotion() {
    this.player.demote();

    if (this.player.demotionsRemaining < 0) {
      this.scene.stop("UIScene");
      this.scene.start("GameOverScene", {
        title: this.player.title,
        score: Math.floor(this.player.score)
      });
      return;
    }

    this.scene.pause();
    this.scene.launch("DemotionScene", {
      title: this.player.title,
      demotionsRemaining: this.player.demotionsRemaining
    });
  }
}

Difficulty manager class

Here’s a clean way to structure it.

export default class DifficultyManager {
  constructor(scene, player) {
    this.scene = scene;
    this.player = player;
    this.band = 1;
    this.struggleUntil = 0;
  }

  getDifficultyBand(score) {
    if (score >= 1800) return 5;
    if (score >= 1100) return 4;
    if (score >= 600) return 3;
    if (score >= 250) return 2;
    return 1;
  }

  update(time) {
    this.band = this.getDifficultyBand(this.player.score);
  }

  enterStruggleMode(time, duration = 5000) {
    this.struggleUntil = time + duration;
  }

  isStruggling(time) {
    return time < this.struggleUntil;
  }

  getConfig(time) {
    const base = DIFFICULTY_BANDS[this.band - 1];
    const config = { ...base };

    if (this.isStruggling(time)) {
      config.spawnIntervalMin *= 1.2;
      config.spawnIntervalMax *= 1.25;
      config.safeLanesMin = Math.max(config.safeLanesMin, 2);
      config.powerupChance += 0.12;
      config.comboChance *= 0.5;
      config.movingHazardChance *= 0.6;
    }

    return config;
  }
}

    this.difficultyManager = difficultyManager;

    this.obstacles = scene.physics.add.group();
    this.powerups = scene.physics.add.group();

    this.nextSpawnAt = 0;
    this.lastChunkY = 0;
    this.lanes = [72, 168, 264, 360];
  }

  update(time, cameraY) {
    const config = this.difficultyManager.getConfig(time);

    if (time >= this.nextSpawnAt) {
      this.spawnChunk(time, cameraY, config);
      this.scheduleNextSpawn(time, config);
    }

    this.cleanup(cameraY);
  }

  scheduleNextSpawn(time, config) {
    const delay = Phaser.Math.Between(
      config.spawnIntervalMin,
      config.spawnIntervalMax
    );

    this.nextSpawnAt = time + delay;
  }

  cleanup(cameraY) {
    const killY = cameraY + 900;

    this.obstacles.children.iterate((obj) => {
      if (obj && obj.y > killY) obj.destroy();
    });

    this.powerups.children.iterate((obj) => {
      if (obj && obj.y > killY) obj.destroy();
    });
  }

  spawnChunk(time, cameraY, config) {
    const chunkY = cameraY - 220;
    const chunkType = this.pickChunkType(config);

    const safeLaneCount = Phaser.Math.Between(
      config.safeLanesMin,
      config.safeLanesMax
    );

    const safeLanes = this.pickSafeLanes(safeLaneCount);

    switch (chunkType) {
      case "calm":
        this.spawnCalmChunk(chunkY, safeLanes, config);
        break;
      case "standard":
        this.spawnStandardChunk(chunkY, safeLanes, config);
        break;
      case "pressure":
        this.spawnPressureChunk(chunkY, safeLanes, config);
        break;
      case "combo":
        this.spawnComboChunk(chunkY, safeLanes, config);
        break;
      case "reward":
        this.spawnRewardChunk(chunkY, safeLanes, config);
        break;
      case "event":
        this.spawnEventChunk(chunkY, safeLanes, config);
        break;
      default:
        this.spawnStandardChunk(chunkY, safeLanes, config);
    }
  }

  pickChunkType(config) {
    const roll = Math.random();

    const weights = this.getChunkWeights(config.band || 1);

    let running = 0;
    for (const entry of weights) {
      running += entry.weight;
      if (roll <= running) return entry.type;
    }

    return "standard";
  }

  getChunkWeights(band) {
    if (band === 1) {
      return [
        { type: "calm", weight: 0.45 },
        { type: "standard", weight: 0.40 },
        { type: "reward", weight: 0.12 },
        { type: "pressure", weight: 0.03 }
      ];
    }

    if (band === 2) {
      return [
        { type: "calm", weight: 0.22 },
        { type: "standard", weight: 0.48 },
        { type: "reward", weight: 0.10 },
        { type: "pressure", weight: 0.16 },
        { type: "combo", weight: 0.04 }
      ];
    }

    if (band === 3) {
      return [
        { type: "calm", weight: 0.12 },
        { type: "standard", weight: 0.42 },
        { type: "reward", weight: 0.10 },
        { type: "pressure", weight: 0.24 },
        { type: "combo", weight: 0.10 },
        { type: "event", weight: 0.02 }
      ];
    }

    if (band === 4) {
      return [
        { type: "calm", weight: 0.08 },
        { type: "standard", weight: 0.34 },
        { type: "reward", weight: 0.10 },
        { type: "pressure", weight: 0.28 },
        { type: "combo", weight: 0.16 },
        { type: "event", weight: 0.04 }
      ];
    }

    return [
      { type: "calm", weight: 0.04 },
      { type: "standard", weight: 0.30 },
      { type: "reward", weight: 0.10 },
      { type: "pressure", weight: 0.28 },
      { type: "combo", weight: 0.22 },
      { type: "event", weight: 0.06 }
    ];
  }

  pickSafeLanes(count) {
    const shuffled = Phaser.Utils.Array.Shuffle([...this.lanes]);
    return shuffled.slice(0, count);
  }

  getUnsafeLanes(safeLanes) {
    return this.lanes.filter((lane) => !safeLanes.includes(lane));
  }

  spawnHazard(x, y, key = "hazard", data = {}) {
    const hazard = this.obstacles.create(x, y, key);
    hazard.setDataEnabled();
    Object.entries(data).forEach(([k, v]) => hazard.setData(k, v));
    hazard.setVelocityY(data.velocityY || 0);
    return hazard;
  }

  spawnPowerup(x, y, key = "powerup", data = {}) {
    const item = this.powerups.create(x, y, key);
    item.setDataEnabled();
    Object.entries(data).forEach(([k, v]) => item.setData(k, v));
    return item;
  }

  maybeSpawnPowerup(chunkY, safeLanes, config) {
    if (Math.random() < config.powerupChance) {
      const lane = Phaser.Utils.Array.GetRandom(safeLanes);
      this.spawnPowerup(lane, chunkY - 24, "coffee", { type: "coffee" });
    }
  }

  spawnCalmChunk(chunkY, safeLanes, config) {
    const unsafe = this.getUnsafeLanes(safeLanes);
    if (unsafe.length > 0 && Math.random() < 0.6) {
      const lane = Phaser.Utils.Array.GetRandom(unsafe);
      this.spawnHazard(lane, chunkY, "buzzword_cloud", { type: "soft" });
    }

    this.maybeSpawnPowerup(chunkY, safeLanes, { ...config, powerupChance: config.powerupChance + 0.10 });
  }

  spawnStandardChunk(chunkY, safeLanes, config) {
    const unsafe = this.getUnsafeLanes(safeLanes);

    unsafe.forEach((lane) => {
      if (Math.random() < 0.7) {
        const key = Phaser.Utils.Array.GetRandom([
          "slack_storm",
          "meeting_block",
          "reply_all",
          "vague_feedback"
        ]);
        this.spawnHazard(lane, chunkY, key, { type: "standard" });
      }
    });

    this.maybeSpawnPowerup(chunkY, safeLanes, config);
  }

  spawnPressureChunk(chunkY, safeLanes, config) {
    const unsafe = this.getUnsafeLanes(safeLanes);

    unsafe.forEach((lane, i) => {
      this.spawnHazard(lane, chunkY + i * 28, "meeting_block", { type: "pressure" });

      if (Math.random() < config.movingHazardChance) {
        this.spawnHazard(lane, chunkY - 20, "slack_storm", {
          type: "moving",
          velocityY: 20
        });
      }
    });

    this.maybeSpawnPowerup(chunkY, safeLanes, { ...config, powerupChance: config.powerupChance * 0.8 });
  }

  spawnComboChunk(chunkY, safeLanes, config) {
    const unsafe = this.getUnsafeLanes(safeLanes);

    if (unsafe.length >= 2) {
      const selected = Phaser.Utils.Array.Shuffle([...unsafe]).slice(0, 2);

      this.spawnHazard(selected[0], chunkY, "glass_ceiling", { type: "combo" });
      this.spawnHazard(selected[1], chunkY + 36, "reorg_wave", { type: "combo" });
    } else if (unsafe.length === 1) {
      this.spawnHazard(unsafe[0], chunkY, "glass_ceiling", { type: "combo" });
    }

    this.maybeSpawnPowerup(chunkY, safeLanes, { ...config, powerupChance: config.powerupChance * 0.7 });
  }

  spawnRewardChunk(chunkY, safeLanes, config) {
    safeLanes.forEach((lane, i) => {
      if (Math.random() < 0.5) {
        const key = i === 0 ? "promotion_letter" : "coffee";
        this.spawnPowerup(lane, chunkY - i * 20, key, { type: key });
      }
    });

    const unsafe = this.getUnsafeLanes(safeLanes);
    if (unsafe.length > 0 && Math.random() < 0.4) {
      const lane = Phaser.Utils.Array.GetRandom(unsafe);
      this.spawnHazard(lane, chunkY + 20, "buzzword_cloud", { type: "soft" });
    }
  }

  spawnEventChunk(chunkY, safeLanes, config) {
    const unsafe = this.getUnsafeLanes(safeLanes);

    unsafe.forEach((lane) => {
      const key = Phaser.Utils.Array.GetRandom([
        "budget_cut",
        "executive_veto",
        "all_hands",
        "kpi_reset"
      ]);
      this.spawnHazard(lane, chunkY, key, { type: "event" });
    });

    this.maybeSpawnPowerup(chunkY, safeLanes, config);

    this.scene.events.emit("world-event", {
      label: Phaser.Utils.Array.GetRandom([
        "Leadership has updated priorities.",
        "An organizational change is underway.",
        "A new strategic direction has been announced."
      ])
    });
  }
}

Think of each pattern as a designed puzzle beat.

Each pattern says:

whether hazards are staggered
what type of motion exists
where the safe route is

lane 2 = right-middle

This should feel like a big moment.

Pattern selection logic

You want the pattern selector to:

respect current difficulty band
avoid repeating the same pattern too often
favor recovery patterns after struggle
favor combo patterns later
Suggested rules
Rule 1: no exact repeat

Do not spawn the same pattern twice in a row.

Rule 2: post-hit relief

After a hit or demotion:

force a calm or reward pattern within the next 1–2 chunks
Rule 3: combo cooldown

Do not spawn two brutal combo patterns back-to-back until late game

Rule 4: event spacing

Event chunks should be rare and separated by several normal chunks

pickPattern(config, struggleMode = false) {
  let candidates = this.patterns.filter((pattern) => {
    return config.band >= pattern.minBand && config.band <= pattern.maxBand;

  if (struggleMode) {
    candidates = candidates.filter((p) =>
      ["coffeeCorridor", "singleAnnoyance", "mirrorAnnoyance"].includes(p.id)
    );
  }

  if (this.lastPatternId) {
    candidates = candidates.filter((p) => p.id !== this.lastPatternId);

function resolvePatternLanes(lanes, safeLaneCount) {
  const shuffled = Phaser.Utils.Array.Shuffle([...lanes]);

  const safe = shuffled.slice(0, safeLaneCount);
  const unsafe = shuffled.slice(safeLaneCount);

  return {
    safe,
    unsafe,
    refs: {
      safeA: safe[0] ?? null,
      safeB: safe[1] ?? null,
      safeC: safe[2] ?? null,
      unsafeA: unsafe[0] ?? null,
      unsafeB: unsafe[1] ?? null,
      unsafeC: unsafe[2] ?? null
    }
  };
}

4. Weighted pattern selection

You want selection to respect:

current difficulty band
struggle mode
recent pattern history
optional tag preferences

Here’s a simple but solid picker.

function weightedPick(items) {
  const total = items.reduce((sum, item) => sum + item.weight, 0);
  let roll = Math.random() * total;

  for (const item of items) {
    roll -= item.weight;
    if (roll <= 0) return item;
  }

  return items[items.length - 1];
}

import Phaser from "phaser";
import { PATTERNS } from "../data/patterns.js";

export default class ObstacleManager {
  constructor(scene, difficultyManager) {
    this.scene = scene;
    this.difficultyManager = difficultyManager;

    this.lanes = [72, 168, 264, 360];

    this.obstacles = scene.physics.add.group();
    this.powerups = scene.physics.add.group();

    this.nextSpawnAt = 0;

    this.lastPatternId = null;
    this.recentPatternIds = [];
  }

  update(time, cameraY) {
    const config = this.difficultyManager.getConfig(time);

    if (time >= this.nextSpawnAt) {
      this.spawnPatternChunk(time, cameraY, config);
      this.scheduleNextSpawn(time, config);
    }

    this.cleanup(cameraY);
  }

  scheduleNextSpawn(time, config) {
    const delay = Phaser.Math.Between(
      config.spawnIntervalMin,
      config.spawnIntervalMax
    );
    this.nextSpawnAt = time + delay;
  }

  cleanup(cameraY) {
    const killY = cameraY + 900;

    this.obstacles.children.iterate((obj) => {
      if (obj && obj.y > killY) obj.destroy();
    });

    this.powerups.children.iterate((obj) => {
      if (obj && obj.y > killY) obj.destroy();
    });
  }

  spawnPatternChunk(time, cameraY, config) {
    const chunkY = cameraY - 220;
    const pattern = this.pickPattern(config, time);

    if (!pattern) return;

    const resolved = this.resolvePattern(pattern);

    this.spawnPattern(pattern, resolved, chunkY, config);

    this.lastPatternId = pattern.id;
    this.recentPatternIds.push(pattern.id);
    if (this.recentPatternIds.length > 4) {
      this.recentPatternIds.shift();
    }

    if (pattern.eventLabel) {
      this.scene.events.emit("world-event", {
        label: pattern.eventLabel
      });
    }
  }

  pickPattern(config, time) {
    const struggling = this.difficultyManager.isStruggling(time);

    let candidates = PATTERNS.filter((pattern) => {
      return config.band >= pattern.minBand && config.band <= pattern.maxBand;
    });

    if (struggling) {
      candidates = candidates.filter((pattern) =>
        pattern.tags?.includes("recovery") ||
        pattern.tags?.includes("calm") ||
        pattern.tags?.includes("reward")
      );
    }

    if (config.comboChance < 0.1) {
      candidates = candidates.filter((pattern) => !pattern.tags?.includes("combo"));
    }

    if (config.eventChance < 0.05) {
      candidates = candidates.filter((pattern) => !pattern.tags?.includes("event"));
    }

    if (this.lastPatternId) {
      candidates = candidates.filter((pattern) => pattern.id !== this.lastPatternId);
    }

    candidates = candidates.filter((pattern) => {
      const repeats = this.recentPatternIds.filter((id) => id === pattern.id).length;
      return repeats < 2;
    });

    if (candidates.length === 0) {
      return null;
    }

    return weightedPick(candidates.map((pattern) => ({
      ...pattern,
      weight: pattern.weight ?? 1
    })));
  }

  resolvePattern(pattern) {
    const shuffled = Phaser.Utils.Array.Shuffle([...this.lanes]);
    const safeCount = Phaser.Math.Clamp(pattern.safeLaneCount ?? 2, 1, this.lanes.length);

    const safe = shuffled.slice(0, safeCount);
    const unsafe = shuffled.slice(safeCount);

    return {
      safe,
      unsafe,
      refs: {
        safeA: safe[0] ?? null,
        safeB: safe[1] ?? null,
        safeC: safe[2] ?? null,
        unsafeA: unsafe[0] ?? null,
        unsafeB: unsafe[1] ?? null,
        unsafeC: unsafe[2] ?? null
      }
    };
  }

  spawnPattern(pattern, resolved, chunkY, config) {
    for (const hazard of pattern.hazards ?? []) {
      const x = this.resolveLaneRef(hazard.laneRef, resolved);
      if (x == null) continue;

      this.spawnHazard(x, chunkY + (hazard.yOffset ?? 0), hazard.key, {
        type: "hazard",
        moving: !!hazard.moving,
        velocityY: hazard.moving ? 20 : 0
      });
    }

    for (const powerup of pattern.powerups ?? []) {
      const chance = powerup.chance ?? 1.0;
      if (Math.random() > chance) continue;

      const x = this.resolveLaneRef(powerup.laneRef, resolved);
      if (x == null) continue;

      this.spawnPowerup(x, chunkY + (powerup.yOffset ?? 0), powerup.key, {
        type: powerup.key
      });
    }

    if (
      (!pattern.powerups || pattern.powerups.length === 0) &&
      Math.random() < config.powerupChance
    ) {
      const x = resolved.safe[0];
      if (x != null) {
        this.spawnPowerup(x, chunkY - 18, "coffee", { type: "coffee" });
      }
    }
  }

  resolveLaneRef(laneRef, resolved) {
    if (!laneRef) return null;

    if (resolved.refs[laneRef] != null) {
      return resolved.refs[laneRef];
    }

    if (laneRef === "safeRandom") {
      return Phaser.Utils.Array.GetRandom(resolved.safe);
    }

    if (laneRef === "unsafeRandom") {
      return Phaser.Utils.Array.GetRandom(resolved.unsafe);
    }

    return null;
  }

  spawnHazard(x, y, key, data = {}) {
    const obj = this.obstacles.create(x, y, key);
    obj.setDataEnabled();

    for (const [k, v] of Object.entries(data)) {
      obj.setData(k, v);
    }

    if (data.velocityY) {
      obj.setVelocityY(data.velocityY);
    }

    return obj;
  }

  spawnPowerup(x, y, key, data = {}) {
    const obj = this.powerups.create(x, y, key);
    obj.setDataEnabled();

    for (const [k, v] of Object.entries(data)) {
      obj.setData(k, v);
    }

    return obj;
  }
}

function weightedPick(items) {
  const total = items.reduce((sum, item) => sum + item.weight, 0);
  let roll = Math.random() * total;

  for (const item of items) {
    roll -= item.weight;
    if (roll <= 0) return item;
  }

  return items[items.length - 1];
}

const STORAGE_KEY = "ladder_climber_leaderboard_v1";
const MAX_ENTRIES = 10;

export default class LeaderboardManager {
  constructor() {
    this.entries = this.loadEntries();
  }

  loadEntries() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return [];
      const parsed = JSON.parse(raw);
      return Array.isArray(parsed) ? parsed : [];
    } catch (error) {
      console.warn("Failed to load leaderboard:", error);
      return [];
    }
  }

  saveEntries() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(this.entries));
    } catch (error) {
      console.warn("Failed to save leaderboard:", error);
    }
  }

  getEntries() {
    return [...this.entries];
  }

  qualifies(run) {
    if (this.entries.length < MAX_ENTRIES) return true;

    const lowest = this.entries[this.entries.length - 1];
    if (!lowest) return true;

    return run.score > lowest.score;
  }

  addEntry(run) {
    const entry = {
      name: this.sanitizeName(run.name),
      score: run.score ?? 0,
      highestTitle: run.highestTitle ?? "Intern (Unpaid)",
      timeSurvivedMs: run.timeSurvivedMs ?? 0,
      demotions: run.demotions ?? 0,
      achievementsUnlocked: run.achievementsUnlocked ?? 0,
      date: run.date ?? new Date().toISOString()
    };

    this.entries.push(entry);
    this.entries.sort((a, b) => b.score - a.score);

    if (this.entries.length > MAX_ENTRIES) {
      this.entries = this.entries.slice(0, MAX_ENTRIES);
    }

    this.saveEntries();
    return entry;
  }

  sanitizeName(name) {
    const cleaned = (name || "ANON")
      .toUpperCase()
      .replace(/[^A-Z0-9 ]/g, "")
      .trim()
      .slice(0, 12);

    return cleaned.length > 0 ? cleaned : "ANON";
  }

  clear() {
    this.entries = [];
    this.saveEntries();
  }
}

      console.warn("Failed to save achievements:", error);
    }
  }

  resetRun() {
    this.runStats = {
      startTimeMs: 0,
      endTimeMs: 0,
      score: 0,
      highestTitle: "Intern (Unpaid)",
      highestTitleIndex: 0,
      demotions: 0,
      laneChanges: 0,
      lastLane: null,
      laneHistory: [],
      hits: 0,
      lastHitTimeMs: 0,
      longestNoHitStreakMs: 0,
      currentNoHitStartMs: 0,
      powerupsCollected: {},
      hazardsDodged: {},
      promotions: 0,
      gameOver: false
    };

    this.runUnlocked = [];
    this.runUnlockedIds = new Set();
  }

  startRun(timeMs) {
    this.resetRun();
    this.runStats.startTimeMs = timeMs;
    this.runStats.currentNoHitStartMs = timeMs;
  }

  updateTime(timeMs) {
    const noHitStreak = timeMs - this.runStats.currentNoHitStartMs;
    if (noHitStreak > this.runStats.longestNoHitStreakMs) {
      this.runStats.longestNoHitStreakMs = noHitStreak;
    }

    this.checkAchievements(timeMs);
  }

  setScore(score) {
    this.runStats.score = score;
  }

  onTitleChanged(title, titleIndex) {
    this.runStats.highestTitle = title;

    if (titleIndex > this.runStats.highestTitleIndex) {
      this.runStats.highestTitleIndex = titleIndex;
      this.runStats.promotions += 1;
    }

    this.checkAchievements(performance.now());
  }

  onDemotion() {
    this.runStats.demotions += 1;
    this.checkAchievements(performance.now());
  }

  onLaneChanged(laneIndex, timeMs) {
    if (this.runStats.lastLane !== null && this.runStats.lastLane !== laneIndex) {
      this.runStats.laneChanges += 1;
    }

    this.runStats.lastLane = laneIndex;
    this.runStats.laneHistory.push({ laneIndex, timeMs });

    if (this.runStats.laneHistory.length > 12) {
      this.runStats.laneHistory.shift();
    }

    this.checkCustomAchievements(timeMs);
    this.checkAchievements(timeMs);
  }

  onHit(timeMs) {
    this.runStats.hits += 1;
    this.runStats.lastHitTimeMs = timeMs;

    const noHitStreak = timeMs - this.runStats.currentNoHitStartMs;
    if (noHitStreak > this.runStats.longestNoHitStreakMs) {
      this.runStats.longestNoHitStreakMs = noHitStreak;
    }

    this.runStats.currentNoHitStartMs = timeMs;
    this.checkAchievements(timeMs);
  }

  onPowerupCollected(key, timeMs) {
    this.runStats.powerupsCollected[key] = (this.runStats.powerupsCollected[key] || 0) + 1;
    this.checkCustomAchievements(timeMs);
    this.checkAchievements(timeMs);
  }

  onGameOver(timeMs) {
    this.runStats.endTimeMs = timeMs;
    this.runStats.gameOver = true;

    const noHitStreak = timeMs - this.runStats.currentNoHitStartMs;
    if (noHitStreak > this.runStats.longestNoHitStreakMs) {
      this.runStats.longestNoHitStreakMs = noHitStreak;
    }

    this.checkAchievements(timeMs);
  }

  getRunSummary() {
    return {
      score: this.runStats.score,
      highestTitle: this.runStats.highestTitle,
      highestTitleIndex: this.runStats.highestTitleIndex,
      demotions: this.runStats.demotions,
      timeSurvivedMs: Math.max(0, this.runStats.endTimeMs - this.runStats.startTimeMs),
      achievementsUnlockedThisRun: [...this.runUnlocked]
    };
  }

  getGlobalUnlockedIds() {
    return Object.keys(this.globalUnlocked).filter((id) => this.globalUnlocked[id]);
  }

  getRunUnlocked() {
    return [...this.runUnlocked];
  }

  isUnlocked(id) {
    return !!this.globalUnlocked[id];
  }

  unlockAchievement(id) {
    if (this.runUnlockedIds.has(id) || this.globalUnlocked[id]) return;

    const achievement = ACHIEVEMENTS.find((a) => a.id === id);
    if (!achievement) return;

    this.globalUnlocked[id] = true;
    this.saveGlobalUnlocked();

    this.runUnlockedIds.add(id);
    this.runUnlocked.push(achievement);

    if (this.scene) {
      this.scene.events.emit("achievement-unlocked", achievement);
    }
  }

  checkAchievements(timeMs) {
    for (const achievement of ACHIEVEMENTS) {
      if (this.globalUnlocked[achievement.id] || this.runUnlockedIds.has(achievement.id)) {
        continue;
      }

      switch (achievement.type) {
        case "highestTitleIndexAtLeast":
          if (this.runStats.highestTitleIndex >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "timeSurvivedAtLeast":
          if ((timeMs - this.runStats.startTimeMs) >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "demotionsAtLeast":
          if (this.runStats.demotions >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "loseWithinMs":
          if (
            this.runStats.gameOver &&
            (this.runStats.endTimeMs - this.runStats.startTimeMs) <= achievement.value
          ) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "laneChangesAtLeast":
          if (this.runStats.laneChanges >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "longestNoHitStreakAtLeast":
          if (this.runStats.longestNoHitStreakMs >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;

        case "powerupCollectedAtLeast":
          if ((this.runStats.powerupsCollected[achievement.key] || 0) >= achievement.value) {
            this.unlockAchievement(achievement.id);
          }
          break;
      }
    }
  }

  checkCustomAchievements(timeMs) {
    if (!this.isUnlocked("circle_back") && this.detectCircleBack(timeMs)) {
      this.unlockAchievement("circle_back");
    }

    if (!this.isUnlocked("thought_leader") && this.detectThoughtLeader()) {
      this.unlockAchievement("thought_leader");
    }
  }

  detectCircleBack(timeMs) {
    const recent = this.runStats.laneHistory.filter((entry) => timeMs - entry.timeMs <= 5000);
    if (recent.length < 4) return false;

    for (let i = 0; i <= recent.length - 4; i++) {
      const a = recent[i].laneIndex;
      const b = recent[i + 1].laneIndex;
      const c = recent[i + 2].laneIndex;
      const d = recent[i + 3].laneIndex;

      if (a === c && b === d && a !== b) {
        return true;
      }
    }

    return false;
  }

  detectThoughtLeader() {
    const directorIndex = 77;
    const coffeeCount = this.runStats.powerupsCollected["coffee"] || 0;
    return this.runStats.highestTitleIndex >= directorIndex && coffeeCount === 0;
  }
}

function weightedPick(items) {
  const total = items.reduce((sum, item) => sum + item.weight, 0);
  let roll = Math.random() * total;

  for (const item of items) {
    roll -= item.weight;
    if (roll <= 0) return item;
  }

  return items[items.length - 1];
}

  });

  if (floorNumber % 10 === 0) {
    const milestoneTemplates = candidates.filter((t) => t.tags.includes("milestone"));
    if (milestoneTemplates.length > 0) {
      return weightedPick(milestoneTemplates);
    }
  }

  if (lastTemplateId) {
    const filtered = candidates.filter((t) => t.id !== lastTemplateId);
    if (filtered.length > 0) candidates = filtered;
  }

  return weightedPick(candidates);
}
8. Floor build function

This is the real useful piece.

export function buildFloorData(floorNumber, zone, templates, lastTemplateId = null) {
  const template = pickTemplateForFloor(floorNumber, templates, lastTemplateId);

  const depthInZone = getDepthInZone(floorNumber, zone);

  const lightingVariant = getLightingVariant(template, depthInZone);
  const clutterLevel = randomRange(template.clutterRange[0], template.clutterRange[1]);
  const hazardDensity = randomRange(template.hazardDensityRange[0], template.hazardDensityRange[1]);

  return {
    floorNumber,
    zone,
    templateId: template.id,
    tags: template.tags,
    lightingVariant,
    clutterLevel,
    hazardDensity,
    rewardBias: getRewardBias(depthInZone),
    special: template.special || null
  };
}

function randomRange(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}
9. Zone depth helpers

You want floors to get meaner inside a zone.

export function getDepthInZone(floorNumber, zone) {
  if (zone === "basement") return floorNumber;
  if (zone === "cubicle") return floorNumber - 50;
  if (zone === "management") return floorNumber - 125;
  if (zone === "executive") return floorNumber - 225;
  return 1;
}

export function getRewardBias(depthInZone) {
  if (depthInZone <= 10) return 0.30;
  if (depthInZone <= 20) return 0.24;
  if (depthInZone <= 35) return 0.16;
  return 0.10;
}

export function getLightingVariant(template, depthInZone) {
  if (template.defaultLighting === "flicker") return "flicker";
  if (depthInZone > 35) return "dim";
  if (depthInZone > 20 && Math.random() < 0.3) return "cool";
  return template.defaultLighting;
}
